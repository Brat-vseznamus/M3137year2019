    \begin{theorem}
        %<*стабилизациязнака>
        Если функция $f : D\to\R$ непрерывна в точке $x_0$ и $f(x_0)\not=0$, то:
        $$\exists V(x_0) : \forall x\in V(x_0)\cap D \quad \sign f(x) = \sign f(x_0)$$
        %</стабилизациязнака>
    \end{theorem}
